while true; do
  song=$(curl -s https://mydora.restorativland.org/song.json)
  echo -e "===\nName: $(echo $song | jq -r '.song.name')\n"
  link=$(echo "https://mydora.restorativland.org/songs$(echo $song | jq -r '.song.path')")
  echo -e "Link: $link\n===\n"
  #mplayer -a52drc 0 $link > /dev/null 2>&1
  # mplayer -a52drc 0 -msglevel all=5 $link
  mpv --no-video $link
done
